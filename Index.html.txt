<script>
const WHATS = "5599999999999";

let produtos = [
    {nome:"Mountain Bike Aro 29", preco:1299, img:"https://via.placeholder.com/300"},
    {nome:"Bike Speed Pro", preco:2499, img:"https://via.placeholder.com/300"},
    {nome:"Bike Urbana", preco:899, img:"https://via.placeholder.com/300"}
];

let carrinho = JSON.parse(localStorage.getItem("carrinho")) || [];

function render(){
    let html="";
    produtos.forEach((p,i)=>{
        html+=`
        <div class="card">
            <img src="${p.img}">
            <div class="card-body">
                <h3>${p.nome}</h3>
                <p class="preco">R$ ${p.preco}</p>
                <button class="btn" onclick="add(${i})">Adicionar</button>
            </div>
        </div>`;
    });
    document.getElementById("produtos").innerHTML=html;
}

function add(i){
    let item = produtos[i];
    let existe = carrinho.find(p=>p.nome===item.nome);

    if(existe) existe.qtd++;
    else carrinho.push({...item, qtd:1});

    salvar();
}

function salvar(){
    localStorage.setItem("carrinho", JSON.stringify(carrinho));
    atualizar();
}

function atualizar(){
    let lista="";
    let total=0;

    carrinho.forEach((i,index)=>{
        let sub=i.preco*i.qtd;
        total+=sub;

        lista+=`
        <div class="item">
        ${i.nome}<br>
        ${i.qtd}x - R$ ${sub}
        </div>`;
    });

    document.getElementById("lista").innerHTML=lista;
    document.getElementById("total").innerText=total;

    let msg="Pedido:%0A";
    carrinho.forEach(i=>{
        msg+=`${i.nome} x${i.qtd}%0A`;
    });
    msg+=`Total: R$ ${total}`;

    document.getElementById("whats").href =
    `https://wa.me/${WHATS}?text=${msg}`;
}

function finalizar(){
    window.open(document.getElementById("whats").href);
}

render();
atualizar();
</script>
